cmake_minimum_required(VERSION 3.8)

##### core3d module #####

project(core3d)

        types.cpp

        glgraphicsdevice.cpp
        glgraphicsdevice.h
        gltypes.cpp
        gltypes.h
        graphicsdevice.cpp
        graphicsdevice.h

target_link_libraries(core3d window)

add_executable(hellotriangle tests/hellotriangle.cpp)
